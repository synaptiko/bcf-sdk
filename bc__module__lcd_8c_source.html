75 void bc_module_lcd_init(bc_module_lcd_framebuffer_t *framebuffer)
76 {
77  _bc_module_lcd_tca9534a_init();
78 
79  bc_spi_init(BC_SPI_SPEED_1_MHZ, BC_SPI_MODE_0);
80 
81  _bc_module_lcd.framebuffer = framebuffer->framebuffer;
82 
83  // Address lines
84  uint8_t line;
85  uint32_t offs;
86  for (line = 0x01, offs = 1; line <= 128; line++, offs += 18)
87  {
88  // Fill the gate line addresses on the exact place in the buffer
89  _bc_module_lcd.framebuffer[offs] = _bc_module_lcd_reverse(line);

91 
92  // Prepare buffer so the background is "white" reflective
93  bc_module_lcd_clear();
94 
95  _bc_module_lcd.task_id = bc_scheduler_register(_bc_module_lcd_task, NULL, _BC_MODULE_LCD_VCOM_PERIOD);

127 void bc_module_lcd_draw_pixel(int x, int y, bool value)
128 {
129  if (x > 127 || y > 127 || x < 0 || y < 0)
130  {
131  return;
132  }
133 
134  int tmp;
135 
136  switch (_bc_module_lcd.rotation)
137  {
138  case BC_MODULE_LCD_ROTATION_90:
139  {
140  tmp = x;
141  x = 127 - y;
142  y = tmp;
143  break;
144  }
145  case BC_MODULE_LCD_ROTATION_180:
146  {
147  x = 127 - x;
148  y = 127 - y;
149  break;
150  }
151  case BC_MODULE_LCD_ROTATION_270:
152  {
153  tmp = y;
154  y = 127 - x;
155  x = tmp;
156  break;
157  }
158  case BC_MODULE_LCD_ROTATION_0:
159  {
160  break;
161  }
162  default:
163  {
164  break;
165  }
166  }
167 
168  // Skip mode byte + addr byte
169  uint32_t byteIndex = 2;
170  // Skip lines
171  byteIndex += y * 18;
172  // Select column byte
173  byteIndex += x / 8;
174 
175  uint8_t bitMask = 1 << (7 - (x % 8));
176 
177  if (!value)
178  {
179  _bc_module_lcd.framebuffer[byteIndex] |= bitMask;
180  }
181  else
182  {
183  _bc_module_lcd.framebuffer[byteIndex] &= ~bitMask;
184  }
185 }

187 int bc_module_lcd_draw_char(int left, int top, uint8_t ch, bool color)
188 {
189  const bc_font_t *font = _bc_module_lcd.font;
190 
191  int w = 0;
192  uint8_t h = 0;
193  uint16_t i;
194  uint16_t x;
195  uint16_t y;
196  uint8_t bytes;
197 
198  for (i = 0; i < font->length; i++)
199  {
200  if (font->chars[i].code == ch)
201  {
202  w = font->chars[i].image->width;
203  h = font->chars[i].image->heigth;
204 
205  bytes = (w + 7) / 8;
206 
207  for (y = 0; y < h; y++)
208  {
209  for (x = 0; x < w; x++)
210  {
211  uint32_t byteIndex = x / 8;
212  byteIndex += y * bytes;
213 
214  uint8_t bitMask = 1 << (7 - (x % 8));
215 
216  if (font->chars[i].image->image[byteIndex] & bitMask)
217  {
218  bc_module_lcd_draw_pixel(left + x, top + y, !color);
219  }
220  else
221  {
222  bc_module_lcd_draw_pixel(left + x, top + y, color);
223  }
224  }
225  }
226  }
227  }
228 
229  return w;
230 }

256 void bc_module_lcd_draw_line(int x0, int y0, int x1, int y1, bool color)
257 {
258  int16_t step = abs(y1 - y0) > abs(x1 - x0);
259  int16_t tmp;
260 
261  if (step)
262  {
263  tmp = x0;
264  x0 = y0;
265  y0 = tmp;
266 
267  tmp = x1;
268  x1 = y1;
269  y1 = tmp;
270  }
271 
272  if (x0 > x1)
273  {
274  tmp = x0;
275  x0 = x1;
276  x1 = tmp;
277 
278  tmp = y0;
279  y0 = y1;
280  y1 = tmp;
281  }
282 
283  int16_t dx = x1 - x0;
284  int16_t dy = abs(y1 - y0);
285 
286  int16_t err = dx / 2;
287  int16_t ystep;
288 
289  ystep = y0 < y1 ? 1 : -1;
290 
291  for (; x0 <= x1; x0++)
292  {
293  if (step)
294  {
295  bc_module_lcd_draw_pixel(y0, x0, color);
296  }
297  else
298  {
299  bc_module_lcd_draw_pixel(x0, y0, color);
300  }
301 
302  err -= dy;
303 
304  if (err < 0)
305  {
306  y0 += ystep;
307  err += dx;
308  }
309  }
310 }

320 // Using Midpoint circle algorithm
321 void bc_module_lcd_draw_circle(int x0, int y0, int radius, bool color)
322 {
323  int x = radius-1;
324  int y = 0;
325  int dx = 1;
326  int dy = 1;
327  int err = dx - (radius << 1);
328 
329  while (x >= y)
330  {
331  bc_module_lcd_draw_pixel(x0 + x, y0 + y, color);
332  bc_module_lcd_draw_pixel(x0 + y, y0 + x, color);
333  bc_module_lcd_draw_pixel(x0 - y, y0 + x, color);
334  bc_module_lcd_draw_pixel(x0 - x, y0 + y, color);
335  bc_module_lcd_draw_pixel(x0 - x, y0 - y, color);
336  bc_module_lcd_draw_pixel(x0 - y, y0 - x, color);
337  bc_module_lcd_draw_pixel(x0 + y, y0 - x, color);
338  bc_module_lcd_draw_pixel(x0 + x, y0 - y, color);
339 
340  if (err <= 0)
341  {
342  y++;
343  err += dy;
344  dy += 2;
345  }
346  if (err > 0)
347  {
348  x--;
349  dx += 2;
350  err += (-radius << 1) + dx;
351  }
352  }
353 }

389 bool bc_module_lcd_update(void)
390 {
391  if (bc_spi_is_ready())
392  {
393  if (!_bc_module_lcd_tca9534a_init())
394  {
395  return false;
396  }
397 
398  if (!bc_tca9534a_write_pin(&_bc_module_lcd.tca9534a, _BC_MODULE_LCD_LED_DISP_CS_PIN, 0))
399  {
400  _bc_module_lcd.is_tca9534a_initialized = false;
401  return false;
402  }
403 
404  _bc_module_lcd.framebuffer[0] = 0x80 | _bc_module_lcd.vcom;
405 
406  if (!bc_spi_async_transfer(_bc_module_lcd.framebuffer, NULL, BC_LCD_FRAMEBUFFER_SIZE, _bc_spi_event_handler, NULL))
407  {
408  if (!bc_tca9534a_write_pin(&_bc_module_lcd.tca9534a, _BC_MODULE_LCD_LED_DISP_CS_PIN, 1))
409  {
410  _bc_module_lcd.is_tca9534a_initialized = false;
411  }
412  return false;
413  }
414 
415  bc_scheduler_plan_relative(_bc_module_lcd.task_id, _BC_MODULE_LCD_VCOM_PERIOD);
416 
417  _bc_module_lcd.vcom ^= 0x40;
418 
419  return true;
420  }
421 
422  return false;
423 }

470 static inline uint8_t _bc_module_lcd_reverse(uint8_t b)
471 {
472  b = (b & 0xf0) >> 4 | (b & 0x0f) << 4;
473  b = (b & 0xcc) >> 2 | (b & 0x33) << 2;
